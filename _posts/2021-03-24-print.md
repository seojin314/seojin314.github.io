### 1. 출력문과 줄바꿈 \n의 활용
#include <stdio.h>
int main()
{
printf("c언어는 누가 만들었나요?\n\n");
printf("데니스 리치가 만들었어요!\n\n");
printf("데니스 리치를 기리기 위해서\n프로그래밍은 hello world를\n출력하는 것으로 시작한다고\n하더군요!");
return 0;
}
